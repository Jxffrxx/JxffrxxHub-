function BigUI:SaveConfig()
if not self.Config.Enabled then return end
local dt={}
for k,v in pairs(self.Flags)do
dt[k]=v
end
local e=HttpService:JSONEncode(dt)
writefile(self.Config.FileName,e)
end
function BigUI:LoadConfig()
if not self.Config.Enabled then return end
if isfile(self.Config.FileName)then
local raw=readfile(self.Config.FileName)
local ok,dec=pcall(function()return HttpService:JSONDecode(raw)end)
if ok and dec then
for k,v in pairs(dec)do
self.Flags[k]=v
end
end
end
end

self.CurrentTheme=th
end

function BigUI:SaveConfig()
 if not self.Config.Enabled then return end
 local data={}
 for f,v in pairs(self.Flags) do
  data[f]=v
 end
 local encoded=HttpService:JSONEncode(data)
 writefile(self.Config.FileName,encoded)
end

function BigUI:LoadConfig()
 if not self.Config.Enabled then return end
 if isfile(self.Config.FileName) then
  local raw=readfile(self.Config.FileName)
  local ok,dec=pcall(function()return HttpService:JSONDecode(raw)end)
  if ok and dec then
   for k,v in pairs(dec) do
    self.Flags[k]=v
   end
  end
 end
end

function BigUI:CreateWindow(sett)
 local pass=AttemptKeySystem(sett)
 if not pass then return end
 local w=BaseWin.new(sett)
 setmetatable(w,{__index=self})
 return w
end
